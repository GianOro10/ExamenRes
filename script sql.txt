CREATE DATABASE examen_db;
USE examen_db;

CREATE TABLE examen (
  id INT PRIMARY KEY AUTO_INCREMENT,
  nombre VARCHAR(100) NOT NULL,
  puntaje_total INT NOT NULL,
  fecha_presentacion DATETIME NOT NULL,
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
);

CREATE TABLE pregunta (
  id INT PRIMARY KEY AUTO_INCREMENT,
  examen_id INT NOT NULL,
  texto VARCHAR(255) NOT NULL,
  opcion_a VARCHAR(255) NOT NULL,
  opcion_b VARCHAR(255) NOT NULL,
  opcion_c VARCHAR(255) NOT NULL,
  opcion_d VARCHAR(255) NOT NULL,
  respuesta VARCHAR(1) NOT NULL,
  puntaje INT NOT NULL,
  FOREIGN KEY (examen_id) REFERENCES examen(id) ON DELETE CASCADE,
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
);

CREATE TABLE estudiante (
  id INT PRIMARY KEY AUTO_INCREMENT,
  nombre VARCHAR(100) NOT NULL,
  edad INT NOT NULL,
  ciudad VARCHAR(100) NOT NULL,
  zona_horaria VARCHAR(100) NOT NULL,
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
);

CREATE TABLE respuesta (
  id INT PRIMARY KEY AUTO_INCREMENT,
  estudiante_id INT NOT NULL,
  pregunta_id INT NOT NULL,
  opcion VARCHAR(1) NOT NULL,
  FOREIGN KEY (estudiante_id) REFERENCES estudiante(id) ON DELETE CASCADE,
  FOREIGN KEY (pregunta_id) REFERENCES pregunta(id) ON DELETE CASCADE,
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
);
